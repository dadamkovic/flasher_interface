187 __STATIC_FORCEINLINE uint32_t __ROR(uint32_t op1, uint32_t op2)
188 {
189  op2 %= 32U;
190  if (op2 == 0U)
191  {
192  return op1;
193  }
194  return (op1 >> op2) | (op1 << (32U - op2));
195 }
